tips标记弹框 Alert弹框风格 confirm弹框类型 iframe加载新页面弹框

参数说明

名称 默认值 说明
type tips 弹框的类型，有：tips,alert,confirm,iframe这几种类型。
title String("这是一个标题") 弹出框的标题
msg String 弹出框的内容部分，可以加上HTML标签
closeBtn true 是否显示关闭按钮 true/false
bgframe true 是否需要背景层 true/false
btnSureText String("确认") 这里是确认的按钮的文字
btnCancelText String("取消") 这里是取消的按钮的文字
quitTime 1500 自动退出的时间间隔，一般和type:'tips'时候配合使用
url String 这里是iframe类型弹框的url链接的地址
clickSure:function(){} 点击确定按钮后执行回调函数
clickCancel:function(){} 点击取消按钮后执行回调函数

例如:

    $(function(){
       $(".type1").jdqLayer({
            "type" : "tips",
            "msg"  :  "这是tips类型的标记框"
        });
        $(".type2").jdqLayer({
            "type" : "alert",
            "msg"  :  "这是alert类型的标记框",
            "closeBtn" : true,
            "bgframe" : true,
            "clickSure" : function(){
                //这里是回调函数
            }
        });
        $(".type3").jdqLayer({
            "type" : "confirm",
            "msg"  :  "这是confirm类型的标记框",
            "closeBtn" : true,
            "bgframe" : true,
            "btnSureText":"不错",
            "btnCancelText":"一般般",
            "clickSure":function(){
                $(".type3").jdqLayer({"type":"tips","msg":"谢谢你的评价！","quitTime":"800"});
            },
            "clickCancel":function(){
                $(".type3").jdqLayer({"type":"tips","msg":"我会努力,总有一天会让你刮目相看！","quitTime":"1500"});
            }
        });
        $(".type4").jdqLayer({
            "type" : "iframe",
            "url"  : "demo1.html",
            "closeBtn" : true,
            "bgframe" : true,
            "width" : "600",
            "height" : "300"
        });
    })
